Assertions

assert.js | should.js| chai.js | expect.js
should

chai.should();

foo.should.be.a('string');
foo.should.equal('bar');
foo.should.have.length(3);
tea.should.have.property('flavors')
  .with.length(3);
assert

var assert = chai.assert;

assert.typeOf(foo, 'string');
assert.equal(foo, 'bar');
assert.lengthOf(foo, 3)
assert.property(tea, 'flavors');
assert.lengthOf(tea.flavors, 3);
expect

var expect = chai.expect;

expect(foo).to.be.a('string');
expect(foo).to.equal('bar');
expect(foo).to.have.length(3);
expect(tea).to.have.property('flavors')
  .with.length(3);

Spies

Provides a pretty good api for spies

You do not have to learn the differencies between stub and/or mock

Looks much more straight forward to work with

Create a spy function


//given
var spy = jasmine.createSpy('spy');

//when
spy('foo');

//then
expect(spy).toHaveBeenCalledWith('foo');
expect(spy).not.toHaveBeenCalledWith('bar');

Spy an existing object


//given
spyOn(jQuery, 'ajax').and.callFake(function () {});

//when
$.ajax(...);

//then
expect(jQuery.ajax).toHaveBeenCalled();

More goodies

  • jasmine.any
  • jasmine.objectContaining

Match anything with jasmine.any


it("matches any value", function() {
  expect({}).toEqual(jasmine.any(Object));
  expect(12).toEqual(jasmine.any(Number));
});

Match something with jasmine.objectContaining


it("matches objects with the expect key/value pairs", function() {
  var foo = {
    a: 1,
    b: 2,
    bar: "baz"
  };
  expect(foo).toEqual(jasmine.objectContaining({
    bar: "baz"
  }));
  expect(foo).not.toEqual(jasmine.objectContaining({
    c: 37
  }));
});
